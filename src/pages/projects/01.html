Semáforo

Este projeto foi o nosso primeiro contato com a ferramenta da AutoDesk, no projeto as luzes verde, vermelho e amarelo ligam e desligam um após o outro imitando o funcionamento de um semáforo. 

// Code Semáforo
void setup() {
  pinMode(7, OUTPUT); //LED VERMELHO
  pinMode(5, OUTPUT); //LED AMARELO
  pinMode(3, OUTPUT); //LED VERDE
}

void loop() {
  digitalWrite(7, HIGH); //LIGA LED VERMELHO
  delay(1000); // ESPERA UM POUCO
  digitalWrite(7, LOW); //DESLIGA O VERMELHO
  delay(1000); // ESPERA UM POUCO

  digitalWrite(5, HIGH); //LIGA LED AMARELO
  delay(1000); // ESPERA UM POUCO
  digitalWrite(5, LOW); //DESLIGA O AMARELO
  delay(1000); // ESPERA UM POUCO

  digitalWrite(3, HIGH); //LIGA LED VERDE
  delay(1000); // ESPERA UM POUCO
  digitalWrite(3, LOW); //DESLIGA O VERDE
  delay(1000); // ESPERA UM POUCO
}
