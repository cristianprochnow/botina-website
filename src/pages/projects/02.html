Luz de Emergência

O segundo projeto realizado foi uma luz de emergência utilizando um photoresistor para ligar uma led quando não tivesse iluminação no ambiente. 

          // Code Semáforo//
  
  
          int Led = 8;
          int Sensor = A0;              
          int Light = 0;                     
          
          void setup()
          {
              pinMode(Led,OUTPUT);       
          }
          
  
          void loop()
          {
            Light = analogRead(Sensor);       
            if(Light<750)
            {               
              digitalWrite(Led,HIGH);
            }
            else
            {                   
              digitalWrite(Led,LOW);
            }                 
          }
